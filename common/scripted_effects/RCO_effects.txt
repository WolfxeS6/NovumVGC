    hidden_effect = {
        if = {
            limit = {
                check_variable = {
                    var = RCO_PosterImage 
                    value = 1
                    compare  = equals
                }
            }
            set_variable = {
                var = RCO_PosterImage
                value = RCO_new_propaganda_campaign
            }
        }
        else_if = {
            limit = {
                check_variable = {
                    var = RCO_PosterImage2
                    value = 1
                    compare  = equals
                }
            }
            set_variable = {

            }
        }
        else = {
            set_variable = {

            }
        }
    }
}
