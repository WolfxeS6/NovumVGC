			}

		}

			change_tag_from = root

		}

		name = angola.1.b

		    factor = 0
	    }

		}

	}
}

country_event = {			#Restore democracy?
	id = angola.3
	title = angola.3.t
	desc = angola.3.d
	picture = GFX_report_event_unita
	
	trigger = {
		tag = UNI
		has_war = no
		ANG = {
			exists = no
		}
		has_government = nationalist
	}	

	fire_only_once = yes

	option = {
		name = angola.3.a
		ai_chance = {
		    factor = 50
	    }
		ANG = {
			set_nationality = {
				target_country = UNI
				character = ANG_holden_roberto
			}
		}
		if = {
			limit = {
				has_country_leader = { ruling_only = no character = UNI_jonas_savimbi }
			}
			add_country_leader_role = {
				character = UNI_jonas_savimbi
				country_leader = {
					ideology = big_tent
				}
				promote_leader = yes
			}
			set_portraits = {
				character = UNI_jonas_savimbi
				civilian = { large ="gfx/leaders/UNI/Portrait_Jonas_Savimbi2.dds" }
			}
		}
		set_politics = {	
			ruling_party = market_liberal
			elections_allowed = yes
		}
		add_popularity = { 
			ideology = market_liberal 
			popularity = 0.2
		}
		add_popularity = { 
			ideology = nationalist 
			popularity = -0.1
		}
	}
	option = {
		name = angola.3.b
		ai_chance = {
		    factor = 50
	    }
		add_political_power = -50
		if = {
			limit = {
				has_country_leader = { ruling_only = no character = UNI_jonas_savimbi }
			}
			add_country_leader_role = {
				character = UNI_jonas_savimbi
				country_leader = {
					ideology = autocrat
				}
				promote_leader = yes
			}
		}
		ANG = {
			set_nationality = {
				target_country = UNI
				character = ANG_holden_roberto
			}
		}
	}
}
