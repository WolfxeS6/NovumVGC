	USA = {
		exists = yes
		has_capitulated = no
		NOT = {
			has_war_with = ROOT
		}
		OR = {
			has_opinion = { target = ROOT value > 15 }
			is_in_faction_with = ROOT
			ROOT = {
				is_subject_of = USA
			}
		}
	}
}
